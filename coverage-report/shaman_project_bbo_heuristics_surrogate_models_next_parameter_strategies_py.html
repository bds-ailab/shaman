29def _norm_pdf(x, mean, sigma): 

30 """ 

31 Compute probability density for Gaussian distribution. 

32 """ 

33 return ( 

34 1 / (np.sqrt(np.pi * 2) * sigma) * np.exp(-((x - mean) ** 2) / (2 * sigma ** 2)) 

35 )

38def _norm_cdf(x, mean, sigma): 

39 """ 

40 Compute distribution function for Gaussian distribution. 

41 """ 

42 return 1 / 2 * (1 + erf((x - mean) / (sigma * np.sqrt(2))))

49def l_bfgs_b_minimizer(func, ranges, **kwargs): 

50 """ 

51 Apply L-BFGS-B algorithm on a function, constrained by the bounds in the range argument. The 

52 function used in the one implemented in the numpy.optimize package. 

53 The initialization of the algorithm is performed by a random choice on the grid. 

54 

55 Args: 

56 func (function): The function to optimize. 

57 ranges (numpy array of numpy arrays): The parameter space. 

58 

59 Returns: 

60 float: The minimum found of the function. 

61 """ 

62 bounds = [(min(range_), max(range_)) for range_ in ranges] 

63 x_0 = uniform_random_draw(1, ranges) 

64 min_ = minimize(func, x0=x_0, method="L-BFGS-B", bounds=bounds) 

65 return min_.x

68def cma_optimizer(func, ranges, sigma=0.5, **kwargs): 

69 """ 

70 Applies the CMA optimizer upon a given function on the grid described in the ranges argument. 

71 The function used for performing optimization in the function available in the cma package. 

72 

73 Args: 

74 func (function): The function to optimize using CMA. 

75 ranges (numpy array of numpy arrays): The parameter grid on which to optimize the function. 

76 sigma (float): The value for the sigma (the step size) 

77 

78 Returns: 

79 float: The minimum of the function. 

80 """ 

81 # compute the minimum and the maximum of each dimension of the grid 

82 mins = list() 

83 maxs = list() 

84 for _range in ranges: 

85 mins.append(min(_range)) 

86 maxs.append(max(_range)) 

87 bound = (mins, maxs) 

88 x_0 = uniform_random_draw(1, ranges) 

89 evolution_strategy = cma.CMAEvolutionStrategy(x_0, sigma, {"bounds": bound}) 

90 evolution_strategy.optimize(func) 

91 return evolution_strategy.result.xbest

98def compute_maximum_probability_improvement(current_optimum, means, stds): 

99 """Given a current optimum, the estimated means and the estimated standard error, return 

100 the value of the maximum probability improvement. If the standard error is estimated to be 0 

101 for a given data point, the expected improvement is set to 0. 

102 

103 Args: 

104 current_optimum (float): The current best parametrization. 

105 means (np.array): A numpy array containing the means of each data point. 

106 stds (np.array): A numpy array containing the standard error of each data point. 

107 """ 

108 flattened_means = means.flatten() 

109 with np.errstate(divide="ignore"): 

110 maximum_improvement = _norm_cdf(current_optimum, flattened_means, stds) 

111 maximum_improvement[np.isnan(maximum_improvement)] = 0.0 

112 return maximum_improvement

115def maximum_probability_improvement(func, ranges, previous_evaluations): 

116 """ 

117 Given a surrogate function that was regressed by a method that estimates both the mean and 

118 the variance of each data point, computes the probability of improvement at each data point 

119 on the grid. This probability is computed using a closed form formula that makes use of the 

120 gaussian density function. It amounts to the distribution a binary variable whose value is 1 if 

121 the current value under evaluation is better than the estimated optimum and 0 if not. 

122 

123 It returns the data point with the highest probability of improvement among all the ranges. 

124 

125 Args: 

126 func (function): the prediction function. 

127 Careful: Must possess an argument return_std that can be set to True. For example, 

128 estimating the function using Gaussian Process via sklearn is a way to proceed. 

129 ranges (numpy array of numpy arrays): the parameter grid to evaluate the function upon. 

130 previous_evaluations (numpy array): the previous evaluations of the function. 

131 

132 Returns: 

133 numpy array: The data point with the highest probability of improvement. 

134 """ 

135 combination_ranges = np.array(np.meshgrid(*ranges)).T.reshape(-1, len(ranges)) 

136 try: 

137 mean, sigma = func(combination_ranges, return_std=True) 

138 except TypeError: 

139 raise TypeError( 

140 "In order to use Expected Improvement, you have to use a regression " 

141 "method which estimates the mean and the standard deviation of the " 

142 "black-box function." 

143 ) 

144 

145 current_optimum = min(previous_evaluations) 

146 maximum_improvement = compute_maximum_probability_improvement( 

147 current_optimum, mean, sigma 

148 ) 

149 return combination_ranges[np.argmax(maximum_improvement)]

152def compute_expected_improvement(current_optimum, means, stds): 

153 """Given a current optimum, the estimated means and the estimated standard error, return 

154 the value of the expected improvement. If the standard error is estimated to be 0 for a given data point, 

155 the expected improvement is set to 0. 

156 

157 Args: 

158 current_optimum (float): The current best parametrization. 

159 means (np.array): A numpy array containing the means of each +data point. 

160 stds (np.array): A numpy array containing the standard error of each data point. 

161 """ 

162 # flattened means, else, memory error 

163 flattened_means = means.flatten() 

164 with np.errstate(divide="ignore"): 

165 expected_imp = (current_optimum - flattened_means) * _norm_cdf(
