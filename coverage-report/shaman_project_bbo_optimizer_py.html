384 def _async_optimization_step(self, parameter): 

385 """ 

386 Performs the optimization step asynchrously, in order to timeit 

387 and interrupt it if it spends too much time computing. 

388 

389 Args: 

390 parameter (numpy array): the parameter to evaluate. 

391 """ 

392 # Set the truncated value as False 

393 truncated = False 

394 # If the max step cost is dynamic, set it to the median value 

395 if callable(self.max_step_cost): 

396 max_step_cost = self.max_step_cost(self.history["fitness"]) 

397 else: 

398 max_step_cost = self.max_step_cost 

399 # Open a separate thread 

400 pool = ThreadPoolExecutor() 

401 computing_thread = pool.submit(self.compute_result, (parameter)) 

402 start_time = time.time() 

403 while not computing_thread.done(): 

404 # If there is no cost function, use the time as default 

405 # Else, use the function passed by the user 

406 if self.step_cost_function: 

407 cost = self.step_cost_function() 

408 else: 

409 cost = time.time() - start_time 

410 if cost < max_step_cost: 

411 time.sleep(0.1) 

412 else: 

413 # End the process 

414 computing_thread.cancel() 

415 # If there is a on_interrupt method of the black-box, 

416 # call it 

417 if self.on_interrupt: 

418 self.on_interrupt() 

419 # Store the value as truncated 

420 truncated = True 

421 break 

422 if truncated: 

423 perf = max_step_cost 

424 else: 

425 perf = computing_thread.result() 

426 

427 # Store the parameters 

428 self._append_parameters(parameter) 

429 # Store the fitness 

430 self._append_fitness(perf) 

431 # Store the value as not truncated 

432 self._append_truncated(truncated)

583 def optimize(self, callbacks=[lambda x: x]): 

584 """ 

585 Performs the optimization process and saves the optimization results as attributes of the 

586 class. 

587 

588 The workflow is the following: 

589 - Choose the initial parameters using the initialization strategy 

590 

591 - While the stop criterion has not been met: 

592 - Apply black box function on current parameter 

593 - Save the parameters and the corresponding fitness in the heuristic history 

594 - launch heuristic with the enhanced history 

595 """ 

596 # start chronometer at beginning of experiment 

597 chronometer_start = time.time() 

598 # Perform initialization 

599 self._initialize(callbacks) 

600 # Store last parameter in current_parameters variable 

601 current_parameters, _ = self._get_best_performance() 

602 # Initialize the loop with one data point 

603 current_parameters = self._select_next_parameters(current_parameters) 

604 # while the stop criterion has not yet been met 

605 while self.stop_rule: 

606 # Perform optimization step, either synchronously or asynchronously 

607 if self.async_optimization: 

608 self._async_optimization_step(current_parameters) 

609 else: 

610 self._optimization_step(current_parameters) 

611 # Add the optimization step as non-init 

612 self._append_init(False) 

613 # Select the next parameter to evaluate 

614 current_parameters = self._select_next_parameters() 

615 # call callback on the history 

616 for callback in callbacks: 

617 callback(self.history) 

618 # update number of iterations 

619 self.nbr_iteration += 1 

620 self.elapsed_time = time.time() - chronometer_start 

621 # Store the best parameters and the best fitness 

622 self.best_parameters_in_grid, self.best_fitness = self._get_best_performance() 

623 # set launched variable to true value 

624 self.launched = True 

625 return self.best_parameters_in_grid

745 @property 

746 def global_exploration_cost(self): 

747 """ 

748 Compute the global exploration cost, i.e. the number of times spent in suboptimal states 

749 (equivalent to the number and the cost of regressions). 

750 

751 Args: 

752 fitness: The vector containing the fitness values. 

753 

754 Returns: 

755 tuple (int, float): The number of times the algorithm was in a suboptimal zone 

756 when a better one had already been found, the performance loss due to those 

757 regressions. 

758 """ 

759 fitness = self.history["fitness"] 

760 minimum = fitness[0] 

761 number_of_states = 0 

762 performance_cost = 0 

763 for fit in fitness: 

764 if fit <= minimum: 

765 minimum = fit 

766 else: 

767 performance_cost += fit - minimum 

768 number_of_states += 1 

769 return number_of_states, performance_cost

771 @property 

772 def local_exploration_cost(self): 

773 """ 

774 Computes the local exploration cost, i.e. the number of times successive operations 

775 resulted in performance loss (regression) and the total loss due to this difference. 

776 

777 Returns: 

778 tuple (int, float): The number of times this phenomena happened, the 

779 performance loss in fitness 

780 """ 

781 fitness = self.history["fitness"] 

782 number_of_states = 0 

783 performance_cost = 0 

784 for i in range(len(fitness) - 1): 

785 # if the algorithm performs worse 

786 if fitness[i] <= fitness[i + 1]: 

787 performance_cost += fitness[i + 1] - fitness[i] 

788 number_of_states += 1 

789 return number_of_states, performance_cost

792 def size_explored_space(self): 

793 """ 

794 Given the visited parameters (a.k.a the different states visited by the algorithm) and 

795 the total numbers of possible states (a.k.a the mutiplication of the length of each 

796 parameter dimension), computes the percentage of the visited space. Also returns the 

797 percentage of static moves (= percentage of states that were observed several times) as 

798 this metric has to be calculated to compute the size of the explored space. 

799 

800 Returns: 

801 tuple (float, int): The 

802 percentage of explored space and the number of duplicated values in the array. 

803 """ 

804 # Compute space size 

805 space_size = 1 

806 for arr in self.parameter_space: 

807 space_size = space_size * len(arr) 

808 # Compute the number of unique parameters 

809 unique_parameters = np.unique(self.history["parameters"], axis=0) 

810 # Compute the number of different visited coordinates 

811 percentage_explored_space = len(unique_parameters) / space_size * 100 

812 # Compute the number of static states 

813 percentage_static_states = ( 

814 (len(self.history["parameters"]) - len(unique_parameters)) 

815 / len(self.history["parameters"]) 

816 * 100 

817 ) 

818 return percentage_explored_space, percentage_static_states
