18class SurrogateModel(Heuristic): 

19 """ 

20 Object that will perform the surrogate modeling. 

21 Inherits from the mother class Heuristic.

24 def __init__(self, regression_model, next_parameter_strategy, *args, **kwargs): 

25 """ 

26 Initializes a Surrogate Model object, which needs two information: the regression or 

27 interpolation model for modeling the function and the acquisition strategy which samples 

28 the next point on the regressed function. 

29 

30 Args: 

31 regression_model (class): the regression model to use. 

32 Compatible with any model that implements methods called .fit or .predict ( 

33 typically anything included in the sklearn package). 

34 

35 function: the strategy to use when selecting new parameters. 

36 """ 

37 super(SurrogateModel, self).__init__(regression_model, next_parameter_strategy) 

38 # Save regression model argument 

39 # Checks for the fit + predict method of regression function 

40 

41 self.regression_model = regression_model() 

42 if not hasattr(self.regression_model, "predict"): 

43 raise AttributeError( 

44 "Regression model argument must have a method named `predict` " 

45 "defined." 

46 ) 

47 if not hasattr(self.regression_model, "fit"): 

48 raise AttributeError( 

49 "Regression model argument must have a method named `fit` " "defined." 

50 ) 

51 

52 # Save next parameter strategy 

53 self.next_parameter_strategy = next_parameter_strategy 

54 # Save parameter and fitness scaler as attributes 

55 self.parameter_scaler = StandardScaler() 

56 self.fitness_scaler = StandardScaler() 

57 

58 @staticmethod 

59 def _build_prediction_function(fitted_regression_model, parameter_scaler): 

60 """ 

61 Given a fitted regression model, returns a function that predicts the value at data point x 

62 for this model. 

63 

64 Args: 

65 fitted_regression_model (object of class regression_model): A fitted object of class 

66 regression_model 

67 parameter_scaler (object of class Scaler): A fitted scaler to transform the parameter 

68 space. 

69 

70 Returns: 

71 function: The function that can be used to predict the value of x. 

72 """ 

73 # define the function 

74 def prediction_function(data_point, *args, **kwargs): 

75 x_arr = np.array(data_point) 

76 try: 

77 scaled_arr = parameter_scaler.transform(x_arr) 

78 return fitted_regression_model.predict(scaled_arr, *args, **kwargs) 

79 except ValueError: 

80 scaled_arr = parameter_scaler.transform(x_arr.reshape(1, -1)) 

81 return fitted_regression_model.predict(scaled_arr, *args, **kwargs) 

82 

83 # return it 

84 return prediction_function

86 def regression_function(self, history): 

87 """ 

88 Fits the regression or interpolation method on the history of the previous evaluations 

89 and returns the associated prediction function. 

90 

91 Args: 

92 history (dict): the history of the optimization, i.e. the tested parameters and the 

93 associated value. Must be a dictionary of the form {"fitness": [---], 

94 "parameters": [---]}: 

95 

96 Returns: 

97 function: the function to use for regression. 

98 """ 

99 # take history and normalize it using standard scaler 

100 scaled_parameters = self.parameter_scaler.fit_transform(history["parameters"]) 

101 scaled_fitness = self.fitness_scaler.fit_transform( 

102 history["fitness"].reshape(-1, 1) 

103 ) 

104 truncated = np.copy(history["truncated"]) 

105 # perform regression and save resulting function as attribute 

106 # try to pass the truncated argument in order to deal with censored data 

107 try: 

108 self.regression_model.fit( 

109 X=scaled_parameters, y=scaled_fitness, truncated=truncated 

110 ) 

111 # if not, fit the model without this parameter 

112 except TypeError: 

113 self.regression_model.fit(X=scaled_parameters, y=scaled_fitness) 

114 return self._build_prediction_function( 

115 self.regression_model, self.parameter_scaler 

116 )

118 def choose_next_parameter(self, history, ranges, *args, **kwargs): 

119 """ 

120 Choose the next parameter by using the function passed as argument for 

121 next_parameter_strategy and use it for acquisition. 

122 

123 Args: 

124 history (dict): the history of the optimization, i.e. the tested parameters and the 

125 associated value. Must be a dictionary of the form {"fitness": [---], 

126 "parameters": [---]} 

127 ranges (numpy arrays of numpy array): the possible values of each parameter dimension. 

128 

129 Returns: 

130 numpy array: The next parameter, i.e. the child born from the 

131 reproduction of the two parents. 

132 """ 

133 # Build prediction function 

134 prediction_function = self.regression_function(history) 

135 # choose next parameter for this function using the strategy given as input 

136 new_parameter = self.next_parameter_strategy( 

137 prediction_function, 

138 ranges=ranges, 

139 previous_evaluations=self.fitness_scaler.transform( 

140 history["fitness"].reshape(-1, 1) 

141 ), 

142 ) 

143 return new_parameter

145 def evaluate_quality(self, history, *args, **kwargs): 

146 """ 

147 Evaluates the quality of the regression model by returning the RMSE of the regression 

148 function on the already evaluated points. The model must have been already fitted in 

149 order to return a score. 

150 

151 Args: 

152 history (dict): the history of the optimization, i.e. the tested parameters and the 

153 associated value. Must be a dictionary of the form {"fitness": [---], 

154 "parameters": [---]} 

155 

156 Returns: 

157 RMSE (float): The value of the RMSE on the evaluated data points. 

158 """ 

159 return self.regression_model.score( 

160 X=history["parameters"], y=history["fitness"] 

161 )
