1""" 

2This module contains different functions that sample some data points from a parameter grid using 

3different strategies for a smart selection. Many techniques and implementations exist from the 

4literature and for now only two of the most common are available in the BBO package: 

5- Random sampling, which simply consist in randomly drawing some parameter from the grid. 

6- Latin Hypercube Sampling, which consist in finding a design that respect the "non collapse" 

7properties of designs (there is no duplicate on any dimension).

20def uniform_random_draw(number_of_parameters, parameter_space): 

21 """ 

22 Draws randomly number_of_parameters among the parameter_space. 

23 

24 Args: 

25 number_of_parameters (int): The number of parameters to draw. 

26 parameter_space (numpy array of numpy arrays): The parameter space, each array 

27 representing a dimension. 

28 

29 Returns: 

30 numpy array of numpy arrays: an array of size number_of_parameters * 

31 number_of_axis containing the parameters. 

32 """ 

33 if parameter_space.shape[0] != 1: 

34 space = parameter_space 

35 else: 

36 space = np.squeeze(parameter_space) 

37 random_draw = [ 

38 np.random.choice(axis, number_of_parameters, replace=True).tolist() 

39 for axis in space 

40 ] 

41 return np.array(random_draw).T

44def latin_hypercube_sampling(number_of_parameters, parameter_space): 

45 """ 

46 Given a parameter space and a number of parameters to draw, draws number_of_parameters 

47 parameter that respect the Latin Hypercube Sampling rule (no parameter have the same 

48 dimension on any axis). The parameters are randomly drawn. 

49 

50 Note that in order to respect this condition, there must be no more 

51 sampled points than the smallest dimension of the parameter space. 

52 

53 Args: 

54 number_of_parameters (int): The number of parameters to draw. 

55 parameter_space (numpy array of numpy arrays): The parameter space, each array 

56 representing a dimension. 

57 

58 Returns: 

59 numpy array of numpy arrays: an array of size number_of_parameters * 

60 number_of_axis containing the parameters. 

61 """ 

62 # assert that the number of parameters is inferior to the maximum size of the parameter space 

63 smallest_size = np.min([len(arr) for arr in parameter_space]) 

64 assert number_of_parameters <= smallest_size, ( 

65 "The number of parameters to be drawn can't be " 

66 "superior to the smallest dimension of the " 

67 "parameter space." 

68 ) 

69 

70 # empty matrix that will contain the parameter values 

71 random_draw = np.full((number_of_parameters, len(parameter_space)), np.nan) 

72 

73 # For each row in the array 

74 for i in range(number_of_parameters): 

75 # for each dimension 

76 for idx, dimension in enumerate(parameter_space): 

77 # randomly draw the first coordinate 

78 random_coordinate = np.random.choice(dimension, 1, replace=False).tolist() 

79 # while the coordinate is already taken 

80 while random_coordinate in random_draw[:, idx]: 

81 random_coordinate = np.random.choice( 

82 dimension, 1, replace=False 

83 ).tolist() 

84 # store value in random_draw array 

85 random_draw[i, idx] = random_coordinate[0] 

86 return random_draw

89def hybrid_lhs_uniform_sampling(number_of_parameters, parameter_space): 

90 """ 

91 Draws number_of_parameters parameter that respect the Latin Hypercube Sampling rule while the 

92 number of parameter does not exceed the number value in the smallest sample. Beyond, a uniform 

93 random sampling is applied 

94 

95 Args: 

96 number_of_parameters (int): The number of parameters to draw. 

97 parameter_space (numpy array of numpy arrays): The parameter space, each array 

98 representing a dimension. 

99 

100 Returns: 

101 numpy array of numpy arrays: an array of size number_of_parameters * 

102 number_of_axis containing the parameters. 

103 """ 

104 n_lhs_param = number_of_parameters 

105 smallest_size = np.min([len(arr) for arr in parameter_space]) 

106 if smallest_size < number_of_parameters: 

107 n_lhs_param = smallest_size 

108 

109 lhs_draw = latin_hypercube_sampling(n_lhs_param, parameter_space) 

110 ur_draw = uniform_random_draw( 

111 max(number_of_parameters - n_lhs_param, 0), parameter_space 

112 ) 

113 return np.append(lhs_draw, ur_draw, axis=0)
